{{/* @component: Language Selector @description: Displays a language switcher using country flags. Shows translations of the current page or all configured languages from all_languages.toml. @params: This component doesn't accept any parameters, it automatically detects available languages and translations. @example: {{ partial "components/navigation/language-selector.html" . }} @note: The component will show either: 1. Page translations if available (Hugo's built-in .Translations) 2. All languages from site.Data.all_languages.languages if configured 3. Nothing if neither is available or the site is monolingual */}} {{ $currentPage := .Page }} {{ $filePath := "" }} {{ if $currentPage.File }} {{ $filePath = $currentPage.File.Path }} {{ end }} {{ $hasMultipleLanguages := gt (len site.Home.AllTranslations) 1 }} {{ $useAllLanguagesData := isset site.Data "all_languages" }}
{{ if $useAllLanguagesData }} {{ with site.Data }} {{ range $code, $langData := .all_languages.languages }} {{ $countryCode := partial "helpers/get-country-code.html" (dict "code" $code) }} {{ $flagUrl := printf "https://flagcdn.com/24x18/%s.png" $countryCode }} {{ $href := partial "helpers/get-language-url.html" (dict "code" $code "langData" $langData "currentPage" $currentPage "filePath" $filePath ) }} {{ partial "components/media/lazyimg.html" (dict "src" $flagUrl "alt" $langData.languageName "width" 24 "height" 18 ) }} {{ end }} {{ end }} {{ else if .Translations }} {{ range .Translations }} {{ $lang := .Language.Lang }} {{ $countryCode := partial "helpers/get-country-code.html" (dict "code" $lang) }} {{ $flagUrl := printf "https://flagcdn.com/24x18/%s.png" $countryCode }} {{ $href := partial "helpers/get-language-url.html" (dict "code" $lang "currentPage" $currentPage "filePath" $filePath "permalink" .Permalink ) }} {{ partial "components/media/lazyimg.html" (dict "src" $flagUrl "alt" .Language.LanguageName "width" 24 "height" 18 ) }} {{ end }} {{ end }}

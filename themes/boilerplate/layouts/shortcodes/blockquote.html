{{/* Blockquote Shortcode This shortcode wraps the quote/blockquote partial template and can be used in three ways: 1. Using named parameters: {{< blockquote name="John Doe" image="/images/john-doe.jpg" text="Your quote text here" >}} 2. Using inner content for the quote: {{< blockquote name="John Doe" image="/images/john-doe.jpg" >}} Your quote text here {{< /blockquote >}} 3. Simple quote without attribution: {{< blockquote >}} Your quote text here {{< /blockquote >}} Parameters: - text: The quote text (optional if using inner content) - name: The author's name (optional) - image: URL to the author's image (optional) */}} {{ $text := .Get "text" | default .Inner }} {{ $name := .Get "name" }} {{ $imageUrl := .Get "image" }} {{ partial "quote/blockquote" (dict "text" $text "name" $name "imageUrl" $imageUrl ) }}
